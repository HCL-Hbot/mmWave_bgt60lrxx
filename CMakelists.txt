cmake_minimum_required(VERSION 3.13)
set(PICO_BOARD pico_w)

# ATTENTION! Pull in SDK (must be before project) *ATTENTION!
include(pico_sdk_import.cmake)
# ATTENTION! Pull in SDK (must be before project) *ATTENTION!

project(BGT60LTR11AIP C CXX ASM)
set(CMAKE_C_STANDARD 11)
set(CMAKE_CXX_STANDARD 17)

pico_sdk_init()

add_executable(BGT60LTR11AIP 
   main.cpp
)

target_link_libraries(BGT60LTR11AIP PRIVATE
   pico_stdlib
   hardware_spi
   pico_cyw43_arch_none 
)

target_include_directories(BGT60LTR11AIP PRIVATE ./bgt60/)

pico_add_extra_outputs(BGT60LTR11AIP)

pico_enable_stdio_usb(BGT60LTR11AIP 1)
pico_enable_stdio_uart(BGT60LTR11AIP 0)
